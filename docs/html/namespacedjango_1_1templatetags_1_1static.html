Join the given path with the STATIC_URL setting.

Usage::

    {% static path [as varname] %}

Examples::

    {% static "myapp/css/base.css" %}
    {% static variable_with_path %}
    {% static "myapp/css/base.css" as admin_base_css %}
    {% static variable_with_path as varname %}
